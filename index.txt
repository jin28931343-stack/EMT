 * 輔助函式：將毫秒轉換為 MM:SS 格式
 * @param {number} ms 毫秒

function formatTime(ms) {
    const totalSeconds = Math.floor(ms / 1000);

/**
 * 渲染得分總結報告
 */
function renderScoreSummary() {
    let totalTimeMs = 0;
    let correctCount = 0;
    
    // 統計每個分類的結果
    const categoryStats = {
        RED: { total: 0, correct: 0 },
        YELLOW: { total: 0, correct: 0 },
        GREEN: { total: 0, correct: 0 },
        BLACK: { total: 0, correct: 0 }
    };
    
    scoreRecords.forEach(record => {
        totalTimeMs += record.triageTimeMs;
        if (record.isCorrect) {
            correctCount++;
        }
        
        // 統計分類總數和正確數
        // 確保使用正確的屬性來取得顏色鍵
        const correctKey = record.correctCategory.color.toUpperCase();
        if (categoryStats[correctKey]) {
            categoryStats[correctKey].total++;
            if (record.isCorrect) {
                categoryStats[correctKey].correct++;
            }
        }
    });

    const totalPatients = scoreRecords.length;
    const avgTimeMs = totalPatients > 0 ? totalTimeMs / totalPatients : 0;
    const overallAccuracy = totalPatients > 0 ? (correctCount / totalPatients) * 100 : 0;

    // 填充時間統計
    document.getElementById('summary-total-time').textContent = formatTime(totalTimeMs);
    document.getElementById('summary-avg-time').textContent = formatTime(avgTimeMs);

    // 填充總體準確度
    document.getElementById('summary-total-patients').textContent = totalPatients;
    document.getElementById('summary-overall-accuracy').textContent = `${overallAccuracy.toFixed(1)}%`;

    // 填充分類詳細表格
    const tableBody = document.getElementById('summary-table-body');
    tableBody.innerHTML = ''; // 清空表格內容

    // 確保順序是紅黃綠黑
    const triageOrder = ['RED', 'YELLOW', 'GREEN', 'BLACK'];

    triageOrder.forEach(key => {
        const category = tg[key];
        const stats = categoryStats[key];
        
        // --- 修正邏輯: 如果 total 為 0，Accuracy 設為 100% ---
        let accuracy = 0;
        if (stats.total === 0) {
            accuracy = 100; // 總數為 0，則正確率為 100%
        } else {
            accuracy = (stats.correct / stats.total) * 100;
        }
        // --- 修正邏輯結束 ---

        const row = document.createElement('tr');
        row.className = 'hover:bg-gray-50';

        const colorClass = category.class;

        row.innerHTML = `
            <td class="px-6 py-4 whitespace-nowrap">
                <div class="flex items-center">
                    <span class="inline-block w-4 h-4 rounded-full ${colorClass} mr-3"></span>
                    <div class="text-sm font-medium text-gray-900">
                        <span class="bilingual-en">${category.en.split(' - ')[0]}</span>
                        <span class="bilingual-zh block text-xs text-gray-500">${category.zh.split('(')[0]}</span>
                    </div>
                </div>
            </td>
            <td class="px-6 py-4 whitespace-nowrap text-center text-sm font-medium text-gray-900">${stats.total}</td>
            <td class="px-6 py-4 whitespace-nowrap text-center text-sm font-medium text-gray-900">${stats.correct}</td>
            <td class="px-6 py-4 whitespace-nowrap text-center text-sm font-medium ${accuracy >= 70 ? 'text-green-600' : 'text-red-600'}">
                ${accuracy.toFixed(1)}%
            </td>
        `;
        tableBody.appendChild(row);
    });

    // 顯示總結報告，隱藏其他區塊
    pcE.classList.add('hidden');
    scoreSummaryEl.classList.remove('hidden');
    mT.classList.remove('hidden'); // 顯示主標題
}
